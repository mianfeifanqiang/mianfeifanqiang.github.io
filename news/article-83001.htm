1.watch监听data内数据的变化
data: {  a: 100 }, watch: {  a(newval, oldVal) {   // 做点什么。。。   console.log(newval, oldVal)  } }
2.watch监听整个对象，deep: true 深度监测
data: {  return {   msg: {    name: 'hahah',    color: 'red'   }  } }  watch: {  msg: {   handler(newValue, oldValue) {    // 做点什么。。。    console.log(newValue)  },  deep: true }
